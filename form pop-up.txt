    // book form pop up java script
    const openButton = document.getElementById("openPopUp");
    const closeButton = document.getElementById("closebtn");
    const bokModal = document.getElementById("bookModal");


    openButton.addEventListener("click", () => {
        bokModal.classList.add("open");
    });

    closeButton.addEventListener("click", () => {
        bokModal.classList.remove("open");
    });

    document.addEventListener("keydown", (event) => {
        if (event.key === "Escape" && bookModal.classList.contains("open")) {
            bookModal.classList.remove("open");
        }

    if (modalBtn) modalBtn.onclick = function() { modal.style.display = "none"; }

            if (event.target == modal) {

            }
        }
